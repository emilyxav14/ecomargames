<script>
const phases=[
{hint:'Espécie marinha associada a anêmonas e recifes tropicais',answer:'PALHACO'},
{hint:'Gigante oceânico cuja dimensão supera qualquer mamífero terrestre',answer:'BALEIA'},
{hint:'Estrutura calcária vital para ecossistemas submarinos tropicais',answer:'CORAL'},
{hint:'Molusco cefalópode reconhecido por inteligência e múltiplos tentáculos',answer:'POLVO'},
{hint:'Predador cartilaginoso no topo de diversas cadeias marinhas',answer:'TUBARAO'}
];

let currentPhase=0;
let score=0;
const keyState={};

const hintEl=document.getElementById('hint');
const gridEl=document.getElementById('grid');
const inputEl=document.getElementById('guessInput');
const messageEl=document.getElementById('message');
const keyboardEl=document.getElementById('keyboard');

function normalizeWord(word){
  return word.toUpperCase().normalize('NFD').replace(/[\u0300-\u036f]/g,'');
}

function buildTitle(){
  const title='LETROMAR';
  const container=document.getElementById('title');

  title.split('').forEach((l,i)=>{
    const span=document.createElement('span');
    span.textContent=l;
    span.style.animationDelay=(i*0.12)+'s';
    container.appendChild(span);
  });
}

buildTitle();

function createFloatingLetters(){
  const letters='ABCDEFGHIJKLMNOPQRSTUVWXYZ';
  const container=document.getElementById('floatingLetters');

  for(let i=0;i<40;i++){
    const el=document.createElement('div');
    el.className='float-letter';
    el.textContent=letters[Math.floor(Math.random()*letters.length)];

    const size=20+Math.random()*60;

    el.style.left=Math.random()*100+'%';
    el.style.fontSize=size+'px';
    el.style.animationDuration=(8+Math.random()*12)+'s';
    el.style.animationDelay=Math.random()*8+'s';

    container.appendChild(el);
  }
}

createFloatingLetters();

function startGame(){
  const splash=document.getElementById('splash');

  splash.classList.add('hide');

  setTimeout(()=>{
    splash.style.display='none';
    document.getElementById('game').style.display='flex';
    buildKeyboard();
    loadPhase();
  },700);
}

function loadPhase(){
  const phase=phases[currentPhase];

  gridEl.innerHTML='';
  hintEl.innerHTML='<b>🌊 Dica:</b> '+phase.hint;

  inputEl.value='';
  messageEl.textContent='';
}

const rows=['QWERTYUIOP','ASDFGHJKL','ZXCVBNM'];

function buildKeyboard(){
  keyboardEl.innerHTML='';

  rows.forEach((row,rowIndex)=>{
    const rowEl=document.createElement('div');
    rowEl.className='keyboardRow';

    if(rowIndex===2){
      rowEl.appendChild(makeKey('⌫',true,()=>{
        inputEl.value=inputEl.value.slice(0,-1);
      }));
    }

    row.split('').forEach(letter=>{
      rowEl.appendChild(makeKey(letter,false,()=>{
        if(inputEl.value.length<10){
          inputEl.value+=letter;
        }
      }));
    });

    if(rowIndex===2){
      rowEl.appendChild(makeKey('↵',true,submitGuess));
    }

    keyboardEl.appendChild(rowEl);
  });
}

function makeKey(label,large,callback){
  const btn=document.createElement('button');
  btn.className='key'+(large?' large':'');
  btn.textContent=label;
  btn.dataset.letter=label;
  btn.onclick=callback;
  return btn;
}

function updateKeyboard(guess,answer){
  guess.split('').forEach((letter,i)=>{
    const prev=keyState[letter]||'none';

    let next;

    if(answer[i]===letter){
      next='correct';
    }
    else if(answer.includes(letter)){
      next='present';
    }
    else{
      next='absent';
    }

    const priority={correct:3,present:2,absent:1,none:0};

    if(priority[next]>priority[prev]){
      keyState[letter]=next;
    }
  });

  document.querySelectorAll('.key').forEach(btn=>{
    const l=btn.dataset.letter;

    if(!l || l.length!==1)return;

    btn.classList.remove('k-correct','k-present','k-absent');

    if(keyState[l]==='correct')btn.classList.add('k-correct');
    else if(keyState[l]==='present')btn.classList.add('k-present');
    else if(keyState[l]==='absent')btn.classList.add('k-absent');
  });
}

function createRow(guess,answer){
  const row=document.createElement('div');
  row.className='row';

  const result=Array(guess.length).fill('absent');
  const used=Array(answer.length).fill(false);

  // VERDES
  for(let i=0;i<guess.length;i++){
    if(i<answer.length && guess[i]===answer[i]){
      result[i]='correct';
      used[i]=true;
    }
  }

  // AMARELOS
  for(let i=0;i<guess.length;i++){
    if(result[i]!=='correct'){
      for(let j=0;j<answer.length;j++){
        if(!used[j] && guess[i]===answer[j]){
          result[i]='present';
          used[j]=true;
          break;
        }
      }
    }
  }

  // BLOCO CONTÍNUO
  const blockMap=Array(guess.length).fill(-1);
  let blockId=0;

  for(let start=0;start<guess.length;start++){
    let bestLength=0;

    for(let len=guess.length-start;len>=2;len--){
      const segment=guess.slice(start,start+len);

      if(answer.includes(segment)){
        bestLength=len;
        break;
      }
    }

    if(bestLength>=2){
      let canPlace=true;

      for(let k=start;k<start+bestLength;k++){
        if(blockMap[k]!==-1){
          canPlace=false;
          break;
        }
      }

      if(canPlace){
        for(let k=start;k<start+bestLength;k++){
          blockMap[k]=blockId;
        }

        blockId++;
      }
    }
  }

  // RENDER
  for(let i=0;i<guess.length;i++){
    const tile=document.createElement('div');
    tile.className='tile '+result[i];
    tile.textContent=guess[i];

    const currentBlock=blockMap[i];
    const connectedLeft=i>0 && blockMap[i-1]===currentBlock && currentBlock!==-1;
    const connectedRight=i<guess.length-1 && blockMap[i+1]===currentBlock && currentBlock!==-1;

    if(currentBlock!==-1){
      if(connectedLeft){
        tile.style.marginLeft='-4px';
        tile.style.borderTopLeftRadius='4px';
        tile.style.borderBottomLeftRadius='4px';
      }

      if(connectedRight){
        tile.style.marginRight='-4px';
        tile.style.borderTopRightRadius='4px';
        tile.style.borderBottomRightRadius='4px';
      }

      if(connectedLeft && connectedRight){
        tile.style.borderRadius='4px';
      }
    }

    row.appendChild(tile);
  }

  gridEl.appendChild(row);
}

function submitGuess(){
  const answer=phases[currentPhase].answer;
  const guess=normalizeWord(inputEl.value.trim());

  if(!guess){
    messageEl.textContent='⚠ Digite uma palavra';
    return;
  }

  createRow(guess,answer);
  updateKeyboard(guess,answer);

  inputEl.value='';

  if(guess===answer){
    score++;
    document.getElementById('score').textContent=score;

    messageEl.textContent='🎉 Palavra correta!';

    setTimeout(()=>{
      currentPhase++;

      Object.keys(keyState).forEach(k=>delete keyState[k]);

      if(currentPhase>=phases.length){
        hintEl.innerHTML='<h2>🏆 Você concluiu o LetroMar!</h2>';
        messageEl.textContent='Pontuação final: '+score+'/'+phases.length;

        inputEl.style.display='none';
        document.querySelector('.sendBtn').style.display='none';
        keyboardEl.style.display='none';
      }
      else{
        buildKeyboard();
        loadPhase();
      }
    },1300);
  }
  else{
    messageEl.textContent='🌊 Continue tentando';
  }
}

inputEl.addEventListener('keydown',e=>{
  if(e.key==='Enter'){
    submitGuess();
  }
});
